/** Every asynchronous surface must support all of these states (design spec §6). */
export const UI_STATES = [
  'loading',
  'empty',
  'partial-error',
  'global-error',
  'slow-network',
  'invalid-data',
  'not-found',
  'unauthorized'
] as const
export type UiState = (typeof UI_STATES)[number]
